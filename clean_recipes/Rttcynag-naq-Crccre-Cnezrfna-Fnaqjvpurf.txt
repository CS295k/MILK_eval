create_ing(ing31, 1 eggplant, seeded and cut lengthwise into 1/4 inch slices)
create_ing(ing32, 1 red bell pepper, sliced into thin strips)
create_ing(ing33, salt and pepper to taste)
create_ing(ing34, 1 French baguette)
create_ing(ing35, 2 ounces soft goat cheese)
create_ing(ing36, 1/4 cup tapenade (olive spread))
create_ing(ing37, 1/4 cup grated Parmesan cheese)

set(t6)
create_tool(t7, baking sheet)

put(ing32, t7)

combine([ing38, ing33], ing39)

combine([ing36, ing42], ing43)
combine([ing40, ing43], ing44)
combine([ing37, ing44], ing45)

cut(ing46, t8, ing47)
