create_ing(ing0, 4 large carrots, cut into 1 inch pieces)
create_ing(ing1, 1/4 large onion, chopped)
create_ing(ing2, 1 quart vegetable broth)
create_ing(ing3, 1/2 cup chopped fresh cilantro)
create_tool(t0, large saucepan.)
combine([ing0, ing1, ing2, ing3], ing4)
put(ing4, t0)
cook(ing4, t0, ing5)

leave(ing5)

cut(ing5, t1, ing6)
serve(ing6)
